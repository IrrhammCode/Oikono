// 3 lines. That's all you need.
contract MyGame is GameReactor {
    constructor() GameReactor("MyGame", GameType.RPG) {
        enablePlugin(AgentPlugin.SPAWN);
        enablePlugin(AgentPlugin.BALANCE);
    }

    function move(uint256 x, uint256 y) external {
        emit PlayerMoved(msg.sender, x, y, xp);
        // ⚡ Agent handles everything!
    }
}
// Connect from any frontend
import { ethers } from "ethers";

const provider = new ethers.WebSocketProvider(
  "wss://dream-rpc.somnia.network/ws"
);

const reactor = new ethers.Contract(
  "0xGameReactor",
  ["event AgentActionTriggered(uint8, uint8, bytes)"],
  provider
);

reactor.on("AgentActionTriggered",
  (type, plugin, data) => {
    console.log(`[Agent] action triggered!`);
});
